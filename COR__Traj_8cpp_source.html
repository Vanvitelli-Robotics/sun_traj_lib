118 /*
119  Get Delta quaterinion, i.e. initial_Q_now
120 */
121 UnitQuaternion COR_Traj::getDeltaQuat(double secs) const
122 {
123  if (_rot_axis[0] == 0.0 && _rot_axis[1] == 0.0 && _rot_axis[2] == 0.0)
124  {
125  return UnitQuaternion();
126  }
127  else
128  {
129  return UnitQuaternion::angvec(_pos_traj.getAngularPosition(secs), _rot_axis);
130  }
131 }
