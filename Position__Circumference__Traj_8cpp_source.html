37 Position_Circumference_Traj::Position_Circumference_Traj(const Vector<3> &r_hat, const Vector<3> &d,
38  const Vector<3> &pi, const Scalar_Traj_Interface &traj_s)
39  : Position_Traj_Interface(NAN, NAN), _traj_s(traj_s.clone())
40 {
41  Vector<3> _r_hat = unit(r_hat);
42 
43  Vector<3> delta = pi - d;
44 
45  if (fabs(delta * r_hat) >= norm(delta) || (norm(r_hat) < 10.0 * std::numeric_limits<double>::epsilon()))
46  {
47  cout << TRAJ_WARN_COLOR "THE CIRCUMFERENCE TRAJ IS A POINT!" CRESET << endl;
48  _c = pi;
49  _R = Identity;
50  _rho = 0.0;
51  return;
52  }
53 
54  _c = d + (delta * r_hat) * r_hat;
55 
56  _rho = norm(pi - _c);
57 
58  _R.T()[0] = unit(pi - _c);
59  _R.T()[2] = _r_hat;
60  _R.T()[1] = _R.T()[2] ^ _R.T()[0];
61 }

63 /*
64  Compute the params of the circumference given the start point, the final
65  point and radius
66  r_hat = normal to the circumference plane
67  pi = initial point
68  pf = final point
69  rho = radius
70  c (return param) = center
71  angle (return param) = angle to use to generate the scalar trajectory
72  sign_plut (optional input) = there are 2 solutions... this bool select the
73  solution
74 */
75 void Position_Circumference_Traj::two_points_2_center(const Vector<3> &r_hat, const Vector<3> &pi, const Vector<3> &pf,
76  double rho, Vector<3> &c, double &angle, bool sign_plus)
77 {
78  double sign = 1.0;
79  if (!sign_plus)
80  {
81  sign = -1.0;
82  }
83 
84  Vector<3> _r_hat = unit(r_hat);
85 
86  c = ((pf + pi) / 2.0) + sign * sqrt(pow(rho, 2) - pow(norm(pf - pi) / 2.0, 2)) * unit(_r_hat ^ (pf - pi));
87 
88  angle = acos((pf - c) * (pi - c) / (norm(pf - c) * norm(pi - c)));
89 }

190 /*
191  Change the reference frame of the trajectory
192  Apply an homogeneous transfrmation matrix to the trajectory
193  new_T_curr is the homog transf matrix of the current frame w.r.t. the new

195 */
196 void Position_Circumference_Traj::changeFrame(const Matrix<4, 4> &new_T_curr)
197 {
198  Vector<4> tmp_v;
199  tmp_v[3] = 1.0;
200 
201  tmp_v.slice<0, 3>() = _c;
202  _c = (new_T_curr * tmp_v).slice<0, 3>();
203  if (!isAPoint())

216 Vector<3> Position_Circumference_Traj::getPosition(double secs) const

231 Vector<3> Position_Circumference_Traj::getVelocity(double secs) const
232 {
233  if (isAPoint())
234  {
235  return Zeros;
236  }
237 
238  double s = _traj_s->getPosition(secs);
239  double s_dot = _traj_s->getVelocity(secs);
240 
241  return _R * makeVector(-_rho * sin(s) * s_dot, _rho * cos(s) * s_dot, 0.0);
242 }

244 /*
245  Get Acceleration at time secs
246 */
247 Vector<3> Position_Circumference_Traj::getAcceleration(double secs) const
248 {
249  if (isAPoint())
250  {
251  return Zeros;
252  }
253 
254  double s = _traj_s->getPosition(secs);
255  double s_dot = _traj_s->getVelocity(secs);
256  double s_2dot = _traj_s->getAcceleration(secs);
257 
258  return _R * makeVector(-_rho * cos(s) * pow(s_dot, 2) - _rho * sin(s) * s_2dot,
259  -_rho * sin(s) * pow(s_dot, 2) + _rho * cos(s) * s_2dot, 0.0);
260 }
