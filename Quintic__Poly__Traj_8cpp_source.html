192 /*
193  Update poly coefficients
194 */
195 void Quintic_Poly_Traj::updateCoefficients()
196 {
197  // This is the effective total time of motion
198  double t = getDuration();
199 
200  if (t < 10.0 * std::numeric_limits<double>::epsilon())
201  {
202  cout << TRAJ_WARN_COLOR "[Quintic_Poly_Traj] WARNING: duration is zero... "
203  "I will fix this..." CRESET
204  << endl;
205  t = QUINTIC_POLY_EPS_TIME;
206  }
207 
208  // These coefficients are known
209  _poly_coeff[5] = _pi;
210  _poly_coeff[4] = _vi;
211  _poly_coeff[3] = _aci / 2.0;
212 
213  // Prepare pow
214  double t_2 = pow(t, 2);
215  double t_3 = pow(t, 3);
216  double t_4 = pow(t, 4);
217  double t_5 = pow(t, 5);
218 
219  // Construct inverse of A ( A*poly_coeff = B )
220  Matrix<3, 3> A_inv = Data(6.0 / t_5, -3.0 / t_4, 1.0 / (2.0 * t_3), -15.0 / t_4, 7.0 / t_3, -1.0 / t_2, 10.0 / t_3,
221  -4.0 / t_2, 1.0 / (2.0 * t));
222  // Construct B
223  Vector<3> B = makeVector(_pf - _pi - _vi * t - (_aci / 2.0) * t_2, _vf - _vi - _aci * t, _acf - _aci);
224 
225  // Calculate poly coeff
226  _poly_coeff.slice<0, 3>() = A_inv * B;
227 
228  // calculate coeff of dp and ddp as polydiff
229  _vel_poly_coeff = polydiff(_poly_coeff);
230  _acc_poly_coeff = polydiff(_vel_poly_coeff);
231 }
