171 /*
172  Get Delta quaterinion, i.e. initial_Q_now
173 */
174 UnitQuaternion Rotation_Const_Axis_Traj::getDeltaQuat(double secs) const
175 {
176  if (_axis[0] == 0.0 && _axis[1] == 0.0 && _axis[2] == 0.0)
177  {
178  return UnitQuaternion();
179  }
180  else
181  {
182  return UnitQuaternion::angvec(_traj_theta->getPosition(secs), _axis);
183  }
184 }
