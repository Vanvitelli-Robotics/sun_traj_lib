88 */
89 double Trapez_Vel_Traj::getPosition(double secs) const
90 {
91  double t = secs - _initial_time;
92 
93  if (t < 0.0)
94  {
95  return _pi;
96  }
97  if (t <= _tc)
98  {
99  return _pi + 0.5 * _ddp * pow(t, 2);
100  }
101  if (t <= (_tc + _tv))
102  {
103  return _pi + _ddp * _tc * (t - 0.5 * _tc);
104  }
105  if (t <= (2.0 * _tc + _tv))
106  {
107  return _pi - 0.5 * _ddp * pow(_tc, 2) + _ddp * _tc * t - 0.5 * _ddp * pow(t - _tc - _tv, 2);
108  }
109  else
110  {
111  return getFinalPosition();
112  }
113 }
